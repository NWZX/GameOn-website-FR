import { checkFormRealtime, checkForm } from "./modules/validation.js";
import { closeModal, launchModal } from "./modules/modal.js";

//#region Leagacy
function editNav() {
  var x = document.getElementById("myTopnav");
  if (x.className === "topnav") {
    x.className += " responsive";
  } else {
    x.className = "topnav";
  }
}
